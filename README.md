## Install

```bash
npm install --save employ-fadoni-now
```

## Usage

```jsx
import React, { Component } from 'react'

import Button from 'employ-fadoni-now'

class Example extends Component {
  render () {
    return (
      <Button 
        btnText="text"
        onClickHandler={(event) console.log(event)}
      />
    )
  }
}
```
